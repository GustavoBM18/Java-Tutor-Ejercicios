import java.util.*;

public class IngresosUniversidad {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in).useLocale(Locale.US);

        int[][] ALUM = new int[ANIOS][CARRERAS];

        // --- Lectura de la matriz ---
        for (int i = 0; i < ANIOS; i++) {
            System.out.println("A√±o " + (i + 1) + ":");
            for (int j = 0; j < CARRERAS; j++) {
                System.out.print("  ALUM[" + (i + 1) + "][" + (j + 1) + "] = ");
                ALUM[i][j] = sc.nextInt(); // >= 0
            }

        // Par√°metros para (b) y (d)
        System.out.print("X (a√±o 1..12): ");

        int T = sc.nextInt();

        if (X < 1 || X > 12 || Y < 1 || Y > 7 || T < 1 || T > 7) {
            System.out.println("Error: X, Y o T fuera de rango.");
            sc.close();

        }

        // (a) Totales por a√±o
        int[] sumaAnio = new int[ANIOS];
        for (int i = 0; i < ANIOS; i++) {
            int suma = 0;

            sumaAnio[i] = suma;
        }

        System.out.println("\n(a) Totales por a√±o:");
        for (int i = 0; i < ANIOS; i++) {

        }

        double porcentaje = (sumaAnio[X - 1] == 0) ? 0.0
            : (ALUM[X - 1][Y - 1] * 100.0) / sumaAnio[X - 1];
        System.out.printf(Locale.US,
            "%n(b) Porcentaje carrera %d en a√±o %d: %.2f%%%n", Y, X, porcentaje);

        // (c) A√±o y carrera con menor ingreso (m√≠nimo global)
        int minVal = Integer.MAX_VALUE, minAnio = -1, minCar = -1;
        for (int i = 0; i < ANIOS; i++) {
            for (int j = 0; j < CARRERAS; j++) {
                if (ALUM[i][j] < minVal) { minVal = ALUM[i][j]; minAnio = i + 1; minCar = j + 1; }
            }
        }
        System.out.printf("(c) Menor ingreso: %d alumnos (a√±o %d, carrera %d)%n",

        // (d) En la carrera T, a√±o con mayor ingreso

        for (int i = 0; i < ANIOS; i++) {
            if (ALUM[i][T - 1] > maxVal) { maxVal = ALUM[i][T - 1]; anioMax = i + 1; }
        }
        System.out.printf("(d) Carrera %d tuvo su mayor ingreso en el a√±o %d (%d alumnos)%n",
                T, anioMax, maxVal);

        sc.close();
        // FIN
    }
}
